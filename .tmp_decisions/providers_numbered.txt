    1: """
    2: AI provider contract layer.
    3: 
    4: This module defines the official abstraction for provider communication and
    5: request execution. Prompt/message construction belongs to prompts.py.
    6: """
    7: 
    8: from __future__ import annotations
    9: 
   10: from abc import ABC, abstractmethod
   11: import json
   12: from typing import Any, Mapping, Sequence
   13: from urllib.error import HTTPError, URLError
   14: from urllib.request import Request, urlopen
   15: 
   16: from django.conf import settings
   17: from django.core.exceptions import ImproperlyConfigured
   18: 
   19: from .prompts import (
   20:     build_generate_store_draft_messages,

   23:     build_regenerate_store_draft_section_messages,

   27: ProviderRawResponse = dict[str, Any]
   28: 
   29: class AIProviderContract(ABC):
   30:     """
   31:     Official provider interface for AI Store Creation workflow.
   32: 
   33:     Contract decision:
   34:     Implementations return raw provider response payloads.
   35:     Parsing/normalization to workflow schema is handled in later layers.
   36: 
   37:     The workflow is store-anchored; the same store_id is reused across calls.
   38:     """
   39: 
   40:     @abstractmethod
   41:     def generate_store_draft(
   42:         self,
   43:         *,
   44:         store_id: int,
   45:         user_store_description: str,
   46:         available_theme_templates: Sequence[str],
   47:     ) -> ProviderRawResponse:
   48:         """
   49:         Execute provider call for initial draft generation.
   50:         Returns raw provider response.
   51:         """
   52:         raise NotImplementedError
   53: 
   54:     @abstractmethod
   55:     def clarify_store_draft(
   56:         self,
   57:         *,
   58:         store_id: int,
   59:         current_draft: Mapping[str, Any],
   60:         prompt: str,
   61:         context: Mapping[str, Any] | None = None,
   62:     ) -> ProviderRawResponse:
   63:         """
   64:         Execute provider call for clarification pass.
   65:         Returns raw provider response.
   66:         """
   67:         raise NotImplementedError
   68: 
   69:     @abstractmethod
   70:     def regenerate_store_draft(
   71:         self,
   72:         *,
   73:         store_id: int,
   74:         original_store_description: str,
   75:         current_draft: Mapping[str, Any],
   76:         clarification_context: Mapping[str, Any] | Sequence[Any] | None = None,
   77:         available_theme_templates: Sequence[str] | None = None,
   78:     ) -> ProviderRawResponse:
   79:         """
   80:         Execute provider call for regeneration pass.
   81:         Returns raw provider response.
   82:         """
   83:         raise NotImplementedError
   84: 
   85:     @abstractmethod
   86:     def regenerate_store_draft_section(
   87:         self,
   88:         *,
   89:         store_id: int,
   90:         target_section: str,
   91:         original_store_description: str,
   92:         current_draft: Mapping[str, Any],
   93:         clarification_context: Mapping[str, Any] | Sequence[Any] | None = None,
   94:         available_theme_templates: Sequence[str] | None = None,
   95:     ) -> ProviderRawResponse:
   96:         """
   97:         Execute provider call for partial section regeneration.
   98:         Returns raw provider response.
   99:         """
  100:         raise NotImplementedError

  103: class OpenAIProviderClient(AIProviderContract):
  104:     """
  105:     Concrete provider client for OpenAI Chat Completions API.
  106: 
  107:     This class is intentionally limited to provider communication only.
  108:     """
  109: 
  110:     API_URL = "https://api.openai.com/v1/chat/completions"
  111: 
  112:     def __init__(self) -> None:
  113:         self.api_key = settings.AI_API_KEY
  114:         self.model_name = settings.AI_MODEL_NAME
  115:         self.timeout = settings.AI_TIMEOUT
  116: 
  117:     def _build_headers(self) -> dict[str, str]:
  118:         if not self.api_key:
  119:             raise ImproperlyConfigured(
  120:                 "AI_API_KEY is not configured. Set it in environment/.env before calling provider."
  121:             )
  122:         return {
  123:             "Authorization": f"Bearer {self.api_key}",
  124:             "Content-Type": "application/json",
  125:         }
  126: 
  127:     def _call_chat_completions(self, messages: list[dict[str, str]]) -> ProviderRawResponse:
  128:         payload = {
  129:             "model": self.model_name,
  130:             "messages": messages,
  131:             # Request JSON-shaped answer from provider; domain parsing remains outside this layer.
  132:             "response_format": {"type": "json_object"},
  133:         }
  134: 
  135:         request = Request(
  136:             url=self.API_URL,
  137:             data=json.dumps(payload).encode("utf-8"),
  138:             headers=self._build_headers(),
  139:             method="POST",
  140:         )
  141: 
  142:         try:
  143:             with urlopen(request, timeout=self.timeout) as response:
  144:                 response_body = response.read().decode("utf-8")
  145:                 return json.loads(response_body)
  146:         except HTTPError as exc:
  147:             error_body = exc.read().decode("utf-8", errors="replace")
  148:             raise RuntimeError(
  149:                 f"Provider HTTP error {exc.code}: {error_body}"
  150:             ) from exc
  151:         except URLError as exc:
  152:             raise RuntimeError(f"Provider connection error: {exc.reason}") from exc
  153: 
  154:     def generate_store_draft(
  155:         self,
  156:         *,
  157:         store_id: int,
  158:         user_store_description: str,
  159:         available_theme_templates: Sequence[str],
  160:     ) -> ProviderRawResponse:
  161:         """
  162:         Execute official generation flow using description + available templates.
  163: 
  164:         `store_id` remains part of the contract for workflow anchoring.
  165:         """
  166:         messages = build_generate_store_draft_messages(
  167:             user_store_description=user_store_description,
  168:             available_theme_templates=available_theme_templates,
  169:         )
  170:         return self._call_chat_completions(messages)
  171: 
  172:     def clarify_store_draft(
  173:         self,
  174:         *,
  175:         store_id: int,
  176:         current_draft: Mapping[str, Any],
  177:         prompt: str,
  178:         context: Mapping[str, Any] | None = None,
  179:     ) -> ProviderRawResponse:
  180:         messages = build_clarify_store_draft_messages(
  181:             store_id=store_id,
  182:             current_draft=current_draft,
  183:             prompt=prompt,
  184:             context=context,
  185:         )
  186:         return self._call_chat_completions(messages)
  187: 
  188:     def regenerate_store_draft(
  189:         self,
  190:         *,
  191:         store_id: int,
  192:         original_store_description: str,
  193:         current_draft: Mapping[str, Any],
  194:         clarification_context: Mapping[str, Any] | Sequence[Any] | None = None,
  195:         available_theme_templates: Sequence[str] | None = None,
  196:     ) -> ProviderRawResponse:
  197:         """
  198:         Execute official full-regeneration flow.
  199: 
  200:         Uses original description + current draft + optional clarification context
  201:         and optional available template names.
  202:         """
  203:         messages = build_regenerate_store_draft_messages(
  204:             store_id=store_id,
  205:             original_store_description=original_store_description,
  206:             current_draft=current_draft,
  207:             clarification_context=clarification_context,
  208:             available_theme_templates=available_theme_templates,
  209:         )
  210:         return self._call_chat_completions(messages)
  211: 
  212:     def regenerate_store_draft_section(
  213:         self,
  214:         *,
  215:         store_id: int,
  216:         target_section: str,
  217:         original_store_description: str,
  218:         current_draft: Mapping[str, Any],
  219:         clarification_context: Mapping[str, Any] | Sequence[Any] | None = None,
  220:         available_theme_templates: Sequence[str] | None = None,
  221:     ) -> ProviderRawResponse:
  222:         """
  223:         Execute official partial-regeneration flow for one target section.
  224:         """
  225:         messages = build_regenerate_store_draft_section_messages(
  226:             store_id=store_id,
  227:             target_section=target_section,
  228:             original_store_description=original_store_description,
  229:             current_draft=current_draft,
  230:             clarification_context=clarification_context,
  231:             available_theme_templates=available_theme_templates,
  232:         )
  233:         return self._call_chat_completions(messages)
  234:

  236: def get_ai_provider_client() -> AIProviderContract:
  237:     """
  238:     Return the default configured AI provider client.
  239:     """
  240:     return OpenAIProviderClient()
