  134: CONSISTENCY RULES

  197: Keep clarification minimal:
  198: - return only 1 to 3 MCQ questions
  199: - keep options short, clear, and mutually distinct

  201: - avoid unnecessary questions if a reasonable draft can already be generated
  202: 
  203: If clarification is needed:
  204: - return a minimal draft structure only
  205: - set `"clarification_needed": true`
  206: - return `clarification_questions` as MCQ objects

  281:

  356: 
  357: def _render_available_theme_templates(available_theme_templates: Sequence[str]) -> str:
  358:     return "\n".join(str(template_name) for template_name in available_theme_templates)
  359: 
  360: 
  361: def build_generate_store_draft_messages(
  362:     *,
  363:     user_store_description: str,
  364:     available_theme_templates: Sequence[str],
  365: ) -> list[ProviderMessage]:
  366:     prompt_text = _APPROVED_BASE_GENERATION_PROMPT.replace(
  367:         "{{available_theme_templates}}",
  368:         _render_available_theme_templates(available_theme_templates),
  369:     )
  370:     return [
  371:         {"role": "system", "content": prompt_text},
  372:         {"role": "user", "content": str(user_store_description)},
  373:     ]
  374: 
  375: 
  376: def build_clarify_store_draft_messages(
  377:     *,
  378:     store_id: int,
  379:     current_draft: Mapping[str, Any],
  380:     prompt: str,
  381:     context: Mapping[str, Any] | None = None,
  382: ) -> list[ProviderMessage]:
  383:     messages: list[ProviderMessage] = [
  384:         {
  385:             "role": "system",
  386:             "content": _APPROVED_CLARIFICATION_ROUND_PROMPT,
  387:         },
  388:         {"role": "user", "content": f"clarification_input: {prompt}"},
  389:         {"role": "user", "content": f"store_id: {store_id}"},
  390:         {
  391:             "role": "user",
  392:             "content": f"current_draft: {json.dumps(dict(current_draft), ensure_ascii=False)}",
  393:         },
  394:     ]
  395:     if context:
  396:         messages.append(
  397:             {
  398:                 "role": "user",
  399:                 "content": f"context: {json.dumps(dict(context), ensure_ascii=False)}",
  400:             }
  401:         )
  402:     return messages
  403: 
  404: 
  405: def build_regenerate_store_draft_messages(
  406:     *,
  407:     store_id: int,
  408:     original_store_description: str,
  409:     current_draft: Mapping[str, Any],
  410:     clarification_context: Mapping[str, Any] | Sequence[Any] | None = None,
  411:     available_theme_templates: Sequence[str] | None = None,
  412: ) -> list[ProviderMessage]:
  413:     system_prompt = _APPROVED_FULL_REGENERATION_PROMPT
  414:     if available_theme_templates is not None and not isinstance(
  415:         available_theme_templates, (str, bytes)
  416:     ):
  417:         system_prompt += (
  418:             "\n\nAllowed theme template names:\n"
  419:             f"{_render_available_theme_templates(available_theme_templates)}"
  420:         )
  421: 
  422:     messages: list[ProviderMessage] = [
  423:         {"role": "system", "content": system_prompt},
  424:         {"role": "user", "content": f"store_id: {store_id}"},
  425:         {"role": "user", "content": f"original_store_description: {original_store_description}"},
  426:         {
  427:             "role": "user",
  428:             "content": f"current_draft: {json.dumps(dict(current_draft), ensure_ascii=False)}",
  429:         },
  430:     ]
  431:     if clarification_context is not None:
  432:         messages.append(
  433:             {
  434:                 "role": "user",
  435:                 "content": f"clarification_context: {json.dumps(clarification_context, ensure_ascii=False)}",
  436:             }
  437:         )
  438:     return messages
  439: 
  440: 
  441: def build_regenerate_store_draft_section_messages(
  442:     *,
  443:     store_id: int,
  444:     target_section: str,
  445:     original_store_description: str,
  446:     current_draft: Mapping[str, Any],
  447:     clarification_context: Mapping[str, Any] | Sequence[Any] | None = None,
  448:     available_theme_templates: Sequence[str] | None = None,
  449: ) -> list[ProviderMessage]:
  450:     system_prompt = _APPROVED_PARTIAL_REGENERATION_PROMPT
  451:     if (
  452:         target_section == "theme"
  453:         and available_theme_templates is not None
  454:         and not isinstance(available_theme_templates, (str, bytes))
  455:     ):
  456:         system_prompt += (
  457:             "\n\nAllowed theme template names:\n"
  458:             f"{_render_available_theme_templates(available_theme_templates)}"
  459:         )
  460: 
  461:     messages: list[ProviderMessage] = [
  462:         {"role": "system", "content": system_prompt},
  463:         {"role": "user", "content": f"store_id: {store_id}"},
  464:         {"role": "user", "content": f"target_section: {target_section}"},
  465:         {"role": "user", "content": f"original_store_description: {original_store_description}"},
  466:         {
  467:             "role": "user",
  468:             "content": f"current_draft: {json.dumps(dict(current_draft), ensure_ascii=False)}",
  469:         },
  470:     ]
  471:     if clarification_context is not None:
  472:         messages.append(
  473:             {
  474:                 "role": "user",
  475:                 "content": f"clarification_context: {json.dumps(clarification_context, ensure_ascii=False)}",
  476:             }
  477:         )
  478:     return messages
